def countResponseTimeRegressions(responseTimes)
  responseTimes.each_with_index.drop(1).count do |time, i|
    time > responseTimes.first(i).sum / responseTimes.first(i).size
  end
end

def mergeHighDefinitionIntervals(intervals)
  return [] if intervals.empty?

  intervals.sort_by! { |interval| interval[0] }

  merged = []
  current_start, current_end = intervals[0]

  intervals.each do |start, end_time|
    if start <= current_end # Overlapping intervals
      current_end = [current_end, end_time].max
    else
      merged << [current_start, current_end]
      current_start = start
      current_end = end_time
    end
  end

  merged << [current_start, current_end]

  merged
end
